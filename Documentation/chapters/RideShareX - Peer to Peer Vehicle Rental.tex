\section{Introduction \& Concept}

\begin{frame}{Problem Statement}
\begin{itemize}
    \item Lack of efficient vehicle sharing platforms
    \item Difficulty in secure booking and verification
    \item No structured approval workflow
\end{itemize}
\end{frame}

\begin{frame}{Motivation}
\begin{itemize}
    \item Increasing demand for shared mobility
    \item Need for trust-based rental ecosystem
    \item Digital transformation in transportation
\end{itemize}
\end{frame}

\begin{frame}{Objectives}
\begin{itemize}
    \item Secure authentication using Google OAuth
    \item Structured booking and approval mechanism
    \item Role-based access control
    \item Scalable and modular system design
\end{itemize}
\end{frame}

\begin{frame}{System Overview}
\begin{itemize}
    \item Web-based vehicle sharing system
    \item User and Admin roles
    \item Booking lifecycle with payment simulation
\end{itemize}
\end{frame}

\begin{frame}{High-Level Architecture}
\begin{itemize}
    \item Presentation Layer – React
    \item Application Layer – Node.js \& Express
    \item Data Layer – MongoDB \& Cloudinary
\end{itemize}
\end{frame}

\section{Architecture \& System Design}

\begin{frame}{System Architecture}
\framesubtitle{Three-Tier Client–Server Architecture}
\begin{itemize}
    \item \textbf{Presentation Layer}
    \begin{itemize}
        \item React frontend
        \item Handles user interaction \& UI
    \end{itemize}
    \vspace{0.3cm}
    \item \textbf{Application Layer}
    \begin{itemize}
        \item Node.js + Express.js backend
        \item Business logic \& API handling
    \end{itemize}
    \vspace{0.3cm}
    \item \textbf{Data Layer}
    \begin{itemize}
        \item MongoDB (Database)
        \item Cloudinary (Image storage)
    \end{itemize}
\end{itemize}
\vspace{0.3cm}
\textcolor{cambridgeblue}{\textbf{Ensures modularity, scalability, and maintainability}}
\end{frame}

\begin{frame}{Major System Modules}
\begin{columns}[T]
\begin{column}{0.48\textwidth}
    \textbf{Authentication Module}
    \begin{itemize}
        \item Google OAuth login
        \item JWT generation \& authorization
    \end{itemize}
    \vspace{0.2cm}
    \textbf{User Management Module}
    \begin{itemize}
        \item Profile management
        \item Verification handling
    \end{itemize}
    \vspace{0.2cm}
    \textbf{Vehicle Management Module}
    \begin{itemize}
        \item Add / Edit vehicles
        \item Availability management
    \end{itemize}
\end{column}
\begin{column}{0.48\textwidth}
    \textbf{Booking Module}
    \begin{itemize}
        \item Booking lifecycle
        \item Owner approval system
    \end{itemize}
    \vspace{0.2cm}
    \textbf{Payment Module}
    \begin{itemize}
        \item Demo payment simulation
        \item Transaction status tracking
    \end{itemize}
    \vspace{0.2cm}
    \textbf{Admin Module}
    \begin{itemize}
        \item User verification
        \item System monitoring
    \end{itemize}
\end{column}
\end{columns}
\end{frame}
